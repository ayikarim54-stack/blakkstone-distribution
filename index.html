This table is what the -- release detail page's new Tracklist section (with per-track preview) reads -- from going forward. Existing releases submitted before this change won't -- have rows here — see the note at the bottom. create table if not exists release_tracks ( id uuid primary key default gen_random_uuid(), release_id uuid not null references releases(id) on delete cascade, track_number int not null, title text not null, isrc text, audio_url text, duration_seconds int, created_at timestamptz not null default now() ); create index if not exists idx_release_tracks_release_id on release_tracks(release_id); create unique index if not exists idx_release_tracks_release_track_num on release_tracks(release_id, track_number); -- If RLS is enabled (recommended), the artist needs to read their own -- release's tracks, and admin needs full access: -- -- alter table release_tracks enable row level security; -- -- create policy "Users can view tracks on their own releases" -- on release_tracks for select -- using ( -- exists ( -- select 1 from releases -- where releases.id = release_tracks.release_id -- and releases.user_id = auth.uid() -- ) -- ); -- -- create policy "Admins can manage all tracks" -- on release_tracks for all -- using ( -- exists ( -- select 1 from profiles -- where profiles.id = auth.uid() and profiles.plan = 'admin' -- ) -- ); -- -- Note on existing releases: releases submitted before this migration only -- have their single audio_url field (track 1 only, tracks 2+ were uploaded -- to storage but never recorded anywhere retrievable — those files are -- effectively orphaned in the 'audio' bucket under each artist's folder). -- There's no way to reconstruct which storage files belonged to which -- multi-track release after the fact from the database alone. The -- Tracklist UI falls back to showing just the single audio_url track for -- any release with no release_tracks rows, so nothing breaks — but those -- older multi-track releases won't show their 2nd/3rd/etc. tracks until -- re-submitted, since that data was never persisted to begin with.
